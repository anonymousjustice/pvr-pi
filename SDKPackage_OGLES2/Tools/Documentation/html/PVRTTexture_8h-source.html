00144 struct MetaDataBlock
00145 {
00146     PVRTuint32  DevFOURCC;      //A 4cc descriptor of the data type's creator. Values equating to values between 'P' 'V' 'R' 0 and 'P' 'V' 'R' 255 will be used by our headers.
00147     PVRTuint32  u32Key;         //A DWORD (enum value) identifying the data type, and thus how to read it.
00148     PVRTuint32  u32DataSize;    //Size of the Data member.
00149     PVRTuint8*  Data;           //Data array, can be absolutely anything, the loader needs to know how to handle it based on DevFOURCC and Key. Use new operator to assign to it.
00150         
00151     /*!***********************************************************************
00152         @Function       MetaDataBlock
00153         @Description    Meta Data Block Constructor
00154     *************************************************************************/
00155     MetaDataBlock() : DevFOURCC(0), u32Key(0), u32DataSize(0), Data(NULL)
00156     {}
00157         
00158     /*!***********************************************************************
00159         @Function       MetaDataBlock
00160         @Description    Meta Data Block Copy Constructor
00161     *************************************************************************/
00162     MetaDataBlock(const MetaDataBlock& rhs)  : DevFOURCC(rhs.DevFOURCC), u32Key(rhs.u32Key), u32DataSize(rhs.u32DataSize)
00163     {
00164         //Copy the data across.
00165         Data = new PVRTuint8[u32DataSize];
00166         for (PVRTuint32 uiDataAmt=0; uiDataAmt<u32DataSize; ++uiDataAmt)
00167         {
00168             Data[uiDataAmt]=rhs.Data[uiDataAmt];
00169         }
00170     }
00171 
00172     /*!***********************************************************************
00173         @Function       ~MetaDataBlock
00174         @Description    Meta Data Block Destructor
00175     *************************************************************************/
00176     ~MetaDataBlock()
00177     {
00178         if (Data) 
00179             delete [] Data;
00180         Data = NULL;
00181     }
00182 
00183     /*!***********************************************************************
00184         @Function       SizeOfBlock
00185         @Return         size_t Size (in a file) of the block.
00186         @Description    Returns the number of extra bytes this will add to any output files.
00187     *************************************************************************/
00188     size_t SizeOfBlock() const
00189     {
00190         return sizeof(DevFOURCC)+sizeof(u32Key)+sizeof(u32DataSize)+u32DataSize;
00191     }

00193     /*!***********************************************************************
00194         @Function       operator=
00195         @Return         MetaDataBlock This MetaDataBlock after the operation.
00196         @Description    Assigns one MetaDataBlock to the other.
00197     *************************************************************************/
00198     MetaDataBlock& operator=(const MetaDataBlock& rhs)
00199     {
00200         if (&rhs==this)
00201             return *this;
00202 
00203         //Remove pre-existing data.
00204         if (Data)
00205             delete [] Data;
00206         Data=NULL;
00207 
00208         //Copy the basic parameters
00209         DevFOURCC=rhs.DevFOURCC;
00210         u32Key=rhs.u32Key;
00211         u32DataSize=rhs.u32DataSize;
00212 
00213         //Copy the data across.
00214         if (rhs.Data)
00215         {
00216             Data = new PVRTuint8[u32DataSize];
00217             for (PVRTuint32 uiDataAmt=0; uiDataAmt<u32DataSize; ++uiDataAmt)
00218             {
00219                 Data[uiDataAmt]=rhs.Data[uiDataAmt];
00220             }
00221         }
00222 
00223         return *this;
